Sub FormatReport()

    With Range("A1:P1")
        .Interior.ColorIndex = 37
        .Font.Bold = True
    End With
    
    With Range("A:B,D:D,F:P")
        .EntireColumn.AutoFit
    End With
    
    Range("A:A,D:D,F:F,I:I,P:P").Select
        With Selection
            .HorizontalAlignment = xlCenter
    End With
    
    Range("A:P").Select
        With Selection
            .VerticalAlignment = xlCenter
    End With
    
    Range("C:C,E:E,J:J").Select
    With Selection
        .ColumnWidth = 53#
        .WrapText = True
    End With

End Sub
